## 制御構文
```python
# if文
score = 85
if score >= 90:
    print("A")
elif score >= 80:
    print("B")
else:
    print("C")

# for文
for i in range(5):
    print(f"Count: {i}")

for fruit in fruits:
    print(f"好きな果物: {fruit}")

# while文
count = 0
while count < 3:
    print(f"回数: {count}")
    count += 1
```

## 関数
```python
# 基本的な関数
def greet(name):
    return f"こんにちは、{name}さん！"

# デフォルト引数
def calculate_area(width, height=10):
    return width * height

# 複数の戻り値
def get_name_age():
    return "山田", 28

name, age = get_name_age()
```

## クラス
```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def introduce(self):
        return f"私は{self.name}です。{self.age}歳です。"
    
    def have_birthday(self):
        self.age += 1

# インスタンス作成
person = Person("佐藤", 25)
print(person.introduce())
```

## エラーハンドリング
```python
try:
    result = 10 / 0
except ZeroDivisionError:
    print("0で割ることはできません")
except Exception as e:
    print(f"エラーが発生しました: {e}")
finally:
    print("処理が完了しました")
```

## リスト内包表記
```python
# 基本的なリスト内包表記
squares = [x**2 for x in range(10)]

# 条件付きリスト内包表記
even_squares = [x**2 for x in range(10) if x % 2 == 0]
```

## ラムダ関数
```python
# ラムダ関数
add = lambda x, y: x + y
print(add(3, 5))  # 8

# map, filter, sortedと組み合わせ
numbers = [1, 2, 3, 4, 5]
doubled = list(map(lambda x: x * 2, numbers))
evens = list(filter(lambda x: x % 2 == 0, numbers))
```
